23   def test_stationarity(series, alpha=0.05):
24       # ADF检验
25       adf_result = adfuller(series.dropna())
26       print(f"ADF Statistic: {adf_result[0]:.4f}")
27       print(f"ADF p-value: {adf_result[1]:.4f}")
28       print("ADF结论: 序列平稳" if adf_result[1] < alpha else "ADF结论: 序列非平稳")
29   
30       # KPSS检验
31       kpss_result = kpss(series.dropna())
32       print(f"\nKPSS Statistic: {kpss_result[0]:.4f}")
33       print(f"KPSS p-value: {kpss_result[1]:.4f}")
34       print("KPSS结论: 序列趋势平稳" if kpss_result[1] > alpha else "KPSS结论: 序列存在单位根")

36   def winsorize_series(series, lower=0.01, upper=0.99):
37       """ 
38       对收益率序列进行缩尾处理 
39       :param series: pandas Series, 收益率序列 
40       :param lower: 下缩尾分位数（默认1%） 
41       :param upper: 上缩尾分位数（默认99%） 
42       :return: 缩尾后的Series 
43       """
44       # 计算分位数边界
45       lower_bound = series.quantile(lower)
46       upper_bound = series.quantile(upper)
47   
48       # 执行缩尾（极端值替换为边界值）
49       return series.clip(lower=lower_bound, upper=upper_bound)

61   def SentimentIndex(df, stock):
62       trade_date = stock['trade_date'].tolist()
63       trade_date = sorted(trade_date)
64       for i in range(len(trade_date)-1):
65           start_time = pd.to_datetime(trade_date[i].strftime('%Y%m%d')+ ' ' + '15:00')
66           end_time = pd.to_datetime((trade_date[i+1].strftime('%Y%m%d') + ' ' + '15:00'))
67           df_sentiment = df[(df['post_datetime']>start_time) & (df['post_datetime']<=end_time)]
68           if df_sentiment.empty:
69               continue
70           else:
71               sentiment_index = (sum(df_sentiment['pos_num']) - sum(df_sentiment['neg_num']))/sum(df_sentiment['word_num'])
72               stock.loc[stock['trade_date'] == pd.to_datetime(trade_date[i+1]).strftime('%Y%m%d'), 'sentiment_index'] = sentiment_index
73               #snownlp_index = (sum(df_sentiment['snownlp'])/len(df_sentiment['snownlp']))
74               #stock.loc[stock['trade_date'] == pd.to_datetime(trade_date[i+1]).strftime('%Y%m%d'), 'snownlp_index'] = snownlp_index
75       return stock
